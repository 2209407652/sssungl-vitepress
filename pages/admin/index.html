import { defineConfig, loadEnv } from "vite";

        },
      },
    },

      host: "0.0.0.0",
      port: VITE_PORT,

登陆鉴权 #

实现思路

  • 登陆: 当用户填写完账号和密码后向服务端验证是否正确，验证通过之后，服务端会返回一个 token(令牌)，拿到 token 之后（我会将这个 token 存贮到 cookie 中，保证刷新页面后能记住用户登录状态），前端会根据 token 再去拉取一个 user_info 的接口来获取用户的详细信息（如用户权限，用户名等等信息）。
  • 权限验证: 通过 token (令牌) 获取用户对应的 role(作用)，动态根据用户的 role (作用)算出其对应有权限的路由，通过 router。addRoutes 动态挂载这些路由。

实现最简单的登陆功能，两个 input 框即可。省略 html 部分代码

async function handleLogin() {
  const { name, password } = loginInfo.value;
  if (!name || !password) {
    ElMessage.warning("请输入用户名和密码");
    return;
  }
  try {
    const res = await loginAPI({ name, password: password.toString() });
    if (res.code === RESPONSE_CODE) {
      ElMessage.success("登录成功");
      setToken(res.data.token); // 登陆成功之后保存 token
      /** isRemember 用来记住用户，登陆成功后才保存在本地 */
      if (isRemember.value) {
        lStorage.set("loginInfo", { name, password });
      } else {
        lStorage.remove("loginInfo");
      }
      /** 登陆成功，跳转到首页 */
      router.push("/");
    } else {
      ElMessage.warning(res.message);
    }
  } catch (error) {
    ElMessage.error(error.message);
  }
}

路由守卫 #

通过路由全局前置钩子来实现动态获取用户信息

js
import { useUserStore } from "@/store/modules/user";
import { usePermissionStore } from "@/store/modules/permission";
import { NOT_FOUND_ROUTE } from "../index";
import { getToken, removeToken } from "@/utils/token";
import { toLogin } from "@/utils/auth";
import { ElMessage } from "element-plus";

const WHITE_LIST = ["/login"];

export function createPermissionGuard(router) {
  const userStore = useUserStore();
  const permissionStore = usePermissionStore();
  /** 路由全局前置守卫钩子函数 */
  router.beforeEach(async (to, from, next) => {
    const token = getToken();
    /**
     * 如果没有 token，那么跳转到 login页面
     * 如果有 token， 则不能跳转到 login 页面。并且有了用户信息则不需要发起请求，没有则请求用户信息保存到 store.user 中
     */
    if (token) {
      if (to.path === "/login") {
        next({ path: "/" });
      } else {
        if (userStore.userId) {
          // 已经拿到了用户信息
          next();
        } else {
          await userStore.getUserInfo().catch((error) => {
            // 请求信息报错后， 清除 token，并且跳转到 login 页面
            removeToken();
            toLogin();
            ElMessage.error(error.message || "获取用户信息失败!");
            return;
          });
          // 根据用户 role(角色), 获取不同权限的路由
          const accessRoutes = permissionStore.generateRoutes(userStore.role);
          accessRoutes.forEach((route) => {
            !router.hasRoute(route.name) && router.addRoute(route);
          });
          router.addRoute(NOT_FOUND_ROUTE);
          next({ ...to, replace: true });
        }
      }
    } else {
      if (WHITE_LIST.includes(to.path)) {
        next();
      } else {
        next({ path: "/login" });
      }
    }
  });
}

pinia #

通过 user 来实现用户信息存储，permission 来实现角色权限控制

permission.js

js
import { defineStore } from "pinia";
import { asyncRoutes } from "@/router";
import { basicRoutes } from "@/router/routes";

function hasPermission(route, role) {
  const routeRole = route.meta?.role ? route.meta.role : [];
  if (!role.length || !routeRole.length) {
    return false;
  }
  return role.some((item) => routeRole.includes(item));
}

function filterAsyncRoutes(routes = [], role) {
  const ret = [];
  routes.forEach((route) => {
    if (hasPermission(route, role)) {
      const curRoute = {
        ...route,
        children: [],
      };
      if (route.children && route.children.length) {
        curRoute.children = filterAsyncRoutes(route.children, role);
      } else {
        Reflect.deleteProperty(curRoute, "children");
      }
      ret.push(curRoute);
    }
  });
  return ret;
}

export const usePermissionStore = defineStore("permission", {
  state() {
    return {
      accessRoutes: [], // 这里保存用户角色应该有权限的路由信息
    };
  },
  getters: {
    routes() {
      return basicRoutes.concat(this.accessRoutes); // concat 用于合拼多个数组，此方法不会改变原数组
    },
    menus() {
      return this.routes.filter((route) => route.name && !route.isHidden);
    },
  },
  actions: {
    generateRoutes(role = []) {
      const accessRoutes = filterAsyncRoutes(asyncRoutes, role);
      this.accessRoutes = accessRoutes;
      return accessRoutes;
    },
  },
});

user.js

import { defineStore } from "pinia";
import { getUserAPI } from "@/api/user";
import { removeToken } from "@/utils/token";
import { toLogin } from "@/utils/auth";
import { RESPONSE_CODE } from "@/utils/request";

export const useUserStore = defineStore("user", {
  state() {
    return {
      userInfo: {},
    };
  },
  getters: {
    userId() {
      return this.userInfo?.id;
    },
    name() {
      return this.userInfo?.name;
    },
    avatar() {
      return this.userInfo?.avatar;
    },
    role() {
      return this.userInfo?.role || [];
    },
  },
  actions: {
    /**
     * 获取用户信息
     * @returns
     */
    async getUserInfo() {
      try {
        const res = await getUserAPI();
        console.log(res);
        if (res.code === RESPONSE_CODE) {
          const { id, name, avatar, role } = res.data;
          this.userInfo = { id, name, avatar, role };

          return Promise.resolve(res.data);
        } else {
          return Promise.reject(res);
        }
      } catch (error) {
        return Promise.reject(error);
      }
    },
    /**
     * 退出登录
     */
    async logout() {
      removeToken();
      this.userInfo = {};
      toLogin();
    },
    /**
     *
     * @param {*} userInfo
     */
    setUserInfo(userInfo = {}) {
      this.userInfo = { ...this.userInfo, ...userInfo };
    },
  },
});

3. 处理路由，获取菜单结果

js
/**
 * 处理路径，外链不处理。
 * 多级菜单需要拼接路径
 */
function resolvePath(basePath, path) {
  // 外部链接直接返回
  if (isExternal(path)) return path;
  return (
    "/" +
    [basePath, path]
      .filter((path) => !!path && path !== "/")
      .map((path) => path.replace(/(^\/)|(\/&)/g, ""))
      .join("/")
  );
}

/**
 * 递归生成多级菜单
 */
function getMenuItem(route, basePath = "") {
  let menuItem = {
    label: (route.meta && route.meta.title) || route.name,
    key: route.name,
    path: resolvePath(basePath, route.path),
    icon: route.meta && route.meta.icon,
  };
  // 有子路由返回有 name 并且 isHidden 不隐藏的子路由，否则返回 []
  const visibleChildren = route.children ? route.children.filter((item) => item.name && !item.isHidden) : [];
  // 没有子路由了直接返回
  if (!visibleChildren.length) return menuItem;
  /** 单个子路由处理 */
  if (visibleChildren.length === 1) {
    const singleRoute = visibleChildren[0];
    menuItem = {
      label: singleRoute.meta?.title || singleRoute.name,
      key: singleRoute.name,
      path: resolvePath(menuItem.path, singleRoute.path),
      icon: singleRoute.meta && singleRoute.meta.icon,
    };
    // 嵌套子路由情况， 递归处理
    const visibleItems = singleRoute.children ? singleRoute.children.filter((item) => item.name && !item.isHidden) : [];
    if (visibleItems.length === 1) {
      menuItem = getMenuItem(visibleItems[0], menuItem.path);
    } else if (visibleItems.length > 1) {
      menuItem.children = visibleItems.map((item) => getMenuItem(item, menuItem.path));
    }
  } else {
    menuItem.children = visibleChildren.map((item) => getMenuItem(item, menuItem.path));
  }

  return menuItem;
}

sidebar-item 组件

vue
<script setup name="sidebar-item">
const props = defineProps({
  route: {
    type: Object,
    default: () => {},
  },
});

function hasChildren(r) {
  if (r.children) return true;
  return false;
}
</script>
<template>
  <!-- 1. 没有子路由的情况 -->
  <el-menu-item v-if="!hasChildren(props.route)" :index="props.route.path">
    <el-icon :class="props.route.icon"></el-icon>
    <template #title>{{ props.route.label }}</template>
  </el-menu-item>
  <!-- 2.有子路由，递归 -->
  <el-sub-menu v-else :index="props.route.path">
    <template #title>
      <el-icon :class="props.route.icon"></el-icon>
      <span>{{ props.route.label }}</span>
    </template>
    <sidebar-item v-for="cr in props.route.children" :key="cr.path" :route="cr" />
  </el-sub-menu>
</template>

el-menu 折叠菜单 bug #

如果一个多级菜单通过递归生成，当 el-menu 折叠时出现 label 和 icon-arrow 两个不会隐藏

解决办法 给 span 标签添加一个类名:

html
<span :class="{ 'el-sub-menu__label': isCollapse }"> {{ props.route.label }} </span>

通过折叠变量 isCollapse 来动态控制类名, css 如下:

scss
.el-sub-menu__label {
  height: 0;
  width: 0;
  overflow: hidden;
  visibility: hidden;
  display: inline-block;

  & + ::v-deep(.el-sub-menu__icon-arrow) {
    display: none;
  }
}
