插件 Flip #

<script lang="ts" setup>
import { nextTick, onMounted, ref } from "vue";
import { gsap } from "gsap";
import Flip from "gsap/dist/Flip";

// all filters items
const allCheckbox = ref<HTMLInputElement>();
let filters: Array<any>, items: Array<any>;

/************ filter array but no flip ************/
function updateFilterNoFlip() {
  const classes = filters
      .filter((checkbox) => checkbox.checked)
      .map((checkbox) => "." + checkbox.id), // filter 筛选选中 ， map 返回选中的 id 数组
    matches = classes.length ? gsap.utils.toArray(classes.join(",")) : classes; // 如果有选中，则获取选中的 item ，否则返回空

  // 调整每个 item 的显示属性, ('none'用于过滤，'inline-flex'用于显示)

  items.forEach((item) => {
    item.style.display = matches.indexOf(item) === -1 ? "none" : "inline-flex";
  });
}

function updateFilter() {
  const state = Flip.getState(items); // 获取当前状态

  updateFilterNoFlip();

  // 从以前的状态添加动画
  Flip.from(state, {
    duration: 0.7,
    scale: true,
    ease: "power1.inOut",
    stagger: 0.08,
    absolute: true,
    onEnter: (elements) =>
      gsap.fromTo(
        elements,
        { opacity: 0, scale: 0 },
        { opacity: 1, scale: 1, duration: 1 }
      ),
    onLeave: (elements) =>
      gsap.to(elements, {
        opacity: 0,
        scale: 0,
        duration: 1,
      }),
  });
}

onMounted(async () => {
  gsap.registerPlugin(Flip);
  // 等待DOM结构刷新
  await nextTick();
  filters = gsap.utils.toArray(".filter");
  items = gsap.utils.toArray(".item");

  // 绑定点击事件
  filters.forEach((btn) => btn.addEventListener("click", updateFilter));
  allCheckbox.value?.addEventListener("click", () => {
    filters.forEach(
      (checkbox) => (checkbox.checked = allCheckbox.value?.checked)
    );
    updateFilter();
  });
});
</script>
